[[extra.phases.steps]]
title = "Moodboard and references"

[[extra.phases.steps]]
title = "Boundaries and scope"
text = "We discuss and document locations, wardrobe, nudity level if any, face visibility, and anything else that needs to be clear in advance."

[[extra.phases.steps]]
title = "Wardrobe and preparation"
text = "We talk through what to bring, how to prepare, and how to arrive as comfortably as possible. There is no pressure to have everything figured out in advance."

[[extra.phases]]
eyebrow = "During the shoot"
title = "The session"

[[extra.phases.steps]]
title = "Your pace"
text = "Sessions are unhurried. We start with time to settle in, because the first part of a shoot is rarely when the best images appear."

[[extra.phases.steps]]
title = "Open communication"
text = "You can pause, adjust, cover up, or stop at any point. Redirecting a pose, changing a setup, or ending early are always available options."

[[extra.phases.steps]]
title = "Privacy on location"
text = "Sessions take place in private spaces: a studio or agreed location. No one else is present unless arranged in advance."

[[extra.phases]]
eyebrow = "After the shoot"
title = "Editing and delivery"

[[extra.phases.steps]]
title = "Selection"
text = "I select a range of the strongest images from the session and prepare them for retouching. You are not required to sift through hundreds of frames."

[[extra.phases.steps]]
title = "Retouching"
text = "Images are edited with care: natural skin texture, soft light, and the character of the original moment are preserved."

[[extra.phases.steps]]
title = "Your review"
text = "You receive the retouched selection to review. You choose what you like, what you want removed, and what may be published."

[[extra.phases.steps]]
title = "Publication approval"
text = "A model release is signed only after your review, covering only the images you have approved. Nothing goes live without this step."

[[extra.faq]]
id = "modeling-experience"
question = "Do I need modeling experience?"
answer = "No. Most collaborators do not have modeling experience. Direction is calm and simple, and there is no expectation to perform or pose in a particular way. The best results come from presence and ease, not technique."

[[extra.faq]]
id = "nude"
question = "Do I have to be nude?"
answer = "No. Boudoir, fine art, and intimate portraiture exist on a wide spectrum. Portraiture, implied nude, anonymous details, silhouette, fabric studies, and partial visibility can all be valid. Nothing is assumed."

[[extra.faq]]
id = "face-private"
question = "Can my face stay private?"
answer = "Yes. Face anonymity can be planned from the beginning through cropping, shadow, angle, fabric, or image selection. Privacy is not something you need to negotiate at the last moment."

[[extra.faq]]
id = "bring-someone"
question = "Can I bring someone with me?"
answer = "Yes, if it helps you feel safe. Please mention this before the shoot so the space and session dynamic work well."

[[extra.faq]]
id = "couples"
question = "Do you work with couples?"
answer = "Yes. Couple sessions can focus on intimacy, tenderness, distance, playfulness, or a more editorial visual story. Both people need to be equally interested, comfortable, and involved in planning the boundaries."

[[extra.faq]]
id = "location"
question = "Where do shoots happen?"
answer = "Shoots happen in Helsinki or nearby, usually in a private interior, studio-like space, or carefully chosen location. The location is discussed and agreed upon before the shoot."

[[extra.faq]]
id = "duration"
question = "How long does a session take?"
answer = "Most sessions take around two to three hours, including time to arrive, talk, settle in, and work without rushing."

[[extra.faq]]
id = "retouch"
question = "Do you retouch images?"
answer = "Yes. Images are edited and lightly retouched, with the goal of atmosphere and polish while preserving natural texture."

[[extra.faq]]
id = "published"
question = "Will my photos be published?"
answer = "Only with written approval. You review the selection, decide which images may be published, and can decline publication without needing to explain why."

[[extra.faq]]
id = "receive"
question = "What do I receive from a TFP shoot?"
answer = "You receive a curated set of edited digital images from the collaboration. The exact number depends on the session and final selection."
